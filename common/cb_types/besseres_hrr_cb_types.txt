			is_emperor = yes
			OR = {
				has_heir = yes
				AND = {
					months_of_ruling = 120
					ruler_age = 30
				}
				AND = {
					months_of_ruling = 60
					NOT = { legitimacy = 40 }
				}
			}

		legitimacy = 80

		NOT = { war_with = FROM }

			NOT = { has_global_flag = hre_golden_bull_enacted }
			has_global_flag = hre_golden_bull_removed
		}
		all_country = {
			is_elected_by_less_hre_electors_than = { who = ROOT }
		}

cb_revoke_electorate_golden_bull = {
	no_opinion_hit = yes

	prerequisites = {
		has_global_flag = hre_golden_bull_enacted
		NOT = { has_global_flag = hre_golden_bull_removed }
		is_emperor = yes
		is_revolutionary = no
		OR = {
			NOT = { has_dlc = "Art of War" }
			hre_leagues_enabled = no
		}
		FROM = {
			is_subject = no
			is_elector = yes
			OR = {
				AND = {
					is_an_ecclesiastical_country = yes
					calc_true_if = {
						all_countries_including_self = {
							is_elector = yes
							is_an_ecclesiastical_country = yes
						}
						amount = 4
					}
					any_country = {
						is_part_of_hre = yes
						is_elector = no
						is_emperor = no
						is_subject = no
						is_a_secular_country = yes
						OR = {
							hre_religion_locked = no
							religion = ROOT
						}
					}
				}
				AND = {
					is_a_secular_country = yes
					calc_true_if = {
						all_countries_including_self = {
							is_elector = yes
							is_a_secular_country = yes
						}
						amount = 5
					}
					any_country = {
						is_part_of_hre = yes
						is_elector = no
						is_emperor = no
						is_subject = no
						is_an_ecclesiastical_country = yes
						OR = {
							hre_religion_locked = no
							religion = ROOT
						}
					}
				}
			}
		}
	}

	attacker_disabled_po = {
		po_demand_provinces
		po_become_vassal
	}

	war_goal = revoke_elector_golden_bull
}
