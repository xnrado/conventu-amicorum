hre_cede_area_to_hre_duchy = {
	[[area]
		$area$ = {
			limit = {
				hre_can_cede_province_to_hre_duchy = { tag = $tag$ }
			}
			owner = {
				add_liberty_desire = 3
				if = {
					limit = {
						is_elector = yes
					}
					set_global_flag = make_hre_duchy_an_elector_$tag$
				}
			}
			if = {
				limit = {
					is_core = owner
				}
				remove_core = owner
				add_core = $tag$
			}
			cede_province = $tag$
			if = {
				limit = {
					NOT = { is_core = $tag$ }
					$tag$ = {
						NOT = { num_of_cities = 2 }
					}
				}
				add_core = $tag$
			}
		}
	]
	[[province]
		$province$ = {
			if = {
				limit = {
					hre_can_cede_province_to_hre_duchy = { tag = $tag$ }
				}
				owner = {
					add_liberty_desire = 3
					if = {
						limit = {
							is_elector = yes
						}
						set_global_flag = make_hre_duchy_an_elector_$tag$
					}
				}
				if = {
					limit = {
						is_core = owner
					}
					remove_core = owner
					add_core = $tag$
				}
				cede_province = $tag$
				if = {
					limit = {
						NOT = { is_core = $tag$ }
						$tag$ = {
							NOT = { num_of_cities = 2 }
						}
					}
					add_core = $tag$
				}
			}
		}
	]
}

hre_set_up_hre_duchy = {
	if = {
		limit = {
			NOT = { is_subject_of = ROOT }
		}
		ROOT = { vassalize = PREV }
		change_government = $government$
		if = {
			limit = {
				government = monarchy
			}
			if = {
				limit = {
					tag = PRU
				}
				add_government_reform = prussian_monarchy
			}
			else_if = {
				limit = {
					tag = HAB
				}
				add_government_reform = austrian_archduchy_reform
			}
			else = {
				add_government_reform = feudalism_reform
			}
		}
		else_if = {
			limit = {
				government = theocracy
			}
			add_government_reform = leading_clergy_reform
		}
		else = {
			change_government = republic
			add_government_reform = oligarchy_reform
		}
		change_religion = ROOT
		change_primary_culture = $culture$
		if = {
			limit = {
				NOT = { dominant_culture = $culture$ }
			}
			add_accepted_culture = new_variable:dominant_culture
		}
		define_ruler = {
			dynasty = ROOT
			claim = 100
		}
		swap_free_idea_group = yes
		swap_non_generic_missions = yes
		add_liberty_desire = -25
		add_treasury = 3000
		add_manpower = 30
		add_adm_power = 200
		add_dip_power = 200
		add_mil_power = 200
		if = {
			limit = {
				NOT = { prestige = 50 }
			}
			add_prestige = 100
		}
		else_if = {
			limit = {
				NOT = { prestige = 25 }
			}
			add_prestige = 50
		}
		else_if = {
			limit = {
				NOT = { prestige = 0 }
			}
			add_prestige = 25
		}
		add_legitimacy = 100
		add_devotion = 100
		add_republican_tradition = 100
	}
	if = {
		limit = {
			OR = {
				has_global_flag = make_hre_duchy_an_elector_@THIS
				NOT = { num_of_electors = 7 }
			}
		}
		clr_global_flag = make_hre_duchy_an_elector_@THIS
		if = {
			limit = {
				is_elector = no
				NOT = { num_of_electors = 7 }
			}
			elector = yes
		}
	}
	add_country_modifier = {
		name = besseres_hrr_hre_duchy_modifier
		duration = -1
	}
	every_province = {
		limit = {
			has_discovered = ROOT
			NOT = { has_discovered = PREV }
		}
		discover_country = PREV
	}
	if = {
		limit = {
			is_part_of_hre = no
		}
		capital_scope = {
			set_in_empire = yes
		}
	}
	every_owned_province = {
		limit = {
			continent = europe
			religion_group = christian
			is_part_of_hre = no
		}
		set_in_empire = yes
	}
	[[kingdom]
		if = {
			limit = {
				always = $kingdom$
				NOT = { government_rank = 2 }
			}
			set_government_rank_force = 2
		}
	]
}
